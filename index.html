files = {
  'data_models.py': 'https://raw.githubusercontent.com/berlinkitchen123-blip/proglove-scanner/main/data_models.py',
  'file_manager.py': 'https://raw.githubusercontent.com/berlinkitchen123-blip/proglove-scanner/main/file_manager.py',

  'json_processor.py': 'https://raw.githubusercontent.com/berlinkitchen123-blip/proglove-scanner/main/json_processor.py',
  'analytics.py': 'https://raw.githubusercontent.com/berlinkitchen123-blip/proglove-scanner/main/analytics.py',
  'main_app.py': 'https://raw.githubusercontent.com/berlinkitchen123-blip/proglove-scanner/main/main_app.py'
}

for filename, url in files.items():
  try:
    response = requests.get(url)
    with open(filename, 'w') as f:
      f.write(response.text)
    print(f"‚úÖ Downloaded {filename}")
  except Exception as e:
    print(f"‚ùå Failed to download {filename}: {e}")

print("üéâ All files downloaded! Run: python main_app.py")
